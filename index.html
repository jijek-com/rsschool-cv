function alphabetPosition(text) {
  const hash = {};
  let str = '';

  for (let i = 0; i < 26; i++) {
    const el = String.fromCharCode((97 + i))
    hash[el] = i + 1
  }

  for (let i = 0; i < text.length; i++) {
    if (text[i].toLowerCase() in hash) {
      str += hash[text[i].toLowerCase()] + ' '
    }
  }

  return str.trim();
}
